/**
 * Hackatón Niño San Borja - API de formulario de interés
 *
 * 1. Crea una hoja de cálculo en Google Sheets.
 * 2. Copia el ID de la hoja desde su URL y pégalo en SPREADSHEET_ID.
 * 3. En Apps Script, implementa como Web App:
 *    - Ejecutar como: Yo
 *    - Quién tiene acceso: Cualquier persona
 * 4. Copia la URL /exec del despliegue a VITE_INTEREST_FORM_ENDPOINT.
 */

const SPREADSHEET_ID = "1UUbeWrhzAqgZxweh2PJV1qIveBacU3UILQAmum-qtzg";
const SHEET_NAME = 'interes243';

const HEADERS = [
  'Fecha y hora',
  'Nombres y apellidos',
  'Correo electrónico',
  'Celular',
  'Perfil',
  'Ciudad o región',
  'Desafíos de interés',
  'Ideas, preguntas u opiniones',
  'Consentimiento',
];

function doGet() {
  return jsonResponse_({ ok: true, service: 'Hackatón Niño San Borja - Formulario de interés' });
}

function doPost(e) {
  try {
    const data = parsePayload_(e);
    validatePayload_(data);

    const lock = LockService.getScriptLock();
    lock.waitLock(10000);

    try {
      const sheet = getOrCreateSheet_();
      ensureHeaders_(sheet);

      sheet.appendRow([
        new Date(),
        clean_(data.fullName),
        clean_(data.email),
        clean_(data.phone),
        clean_(data.profile),
        clean_(data.city),
        Array.isArray(data.interests) ? data.interests.join(' | ') : clean_(data.interests),
        clean_(data.comments),
        data.consent === true ? 'Sí' : 'No',
      ]);
    } finally {
      lock.releaseLock();
    }

    return jsonResponse_({ ok: true });
  } catch (error) {
    console.error(error);
    return jsonResponse_({ ok: false, error: String(error.message || error) });
  }
}

function getOrCreateSheet_() {
  if (!SPREADSHEET_ID) {
    throw new Error('Configura SPREADSHEET_ID con el ID de tu Google Sheet.');
  }

  const spreadsheet = SpreadsheetApp.openById(SPREADSHEET_ID);
  return spreadsheet.getSheetByName(SHEET_NAME) || spreadsheet.insertSheet(SHEET_NAME);
}

function ensureHeaders_(sheet) {
  if (sheet.getLastRow() !== 0) return;
  sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]);
  sheet.getRange(1, 1, 1, HEADERS.length).setFontWeight('bold');
  sheet.setFrozenRows(1);
  sheet.autoResizeColumns(1, HEADERS.length);
}

function parsePayload_(e) {
  const contents = e && e.postData && e.postData.contents;
  if (contents) return JSON.parse(contents);
  return (e && e.parameter) || {};
}

function validatePayload_(data) {
  if (!data.fullName || !data.email || !data.profile) {
    throw new Error('Faltan datos obligatorios.');
  }
  if (!data.consent) {
    throw new Error('Se requiere consentimiento para registrar los datos.');
  }
}

function clean_(value) {
  return String(value || '').trim();
}

function jsonResponse_(payload) {
  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}
